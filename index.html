#Obtaining weather data from a downloaded csv file and reading it into dataframe
import pandas as pd 
import seaborn as sns
import matplotlib.pyplot as plt

weather = pd.read_csv('originalweather.csv', sep = '\t',)
weather = weather.drop([0, 1, 2, 3]) # Drops rows irrelevant to our data
weather = weather.reset_index(drop=True) # Resets the index after dropping
weather.columns = ['Year', 'Temp', 'Anomaly'] # Rename the columns

for i, j in weather.iterrows():
    year = int(j[0][:-2])
    temp = float(j[1])
    weather.at[i, 'Year'] = year
    weather.at[i, 'Temp'] = temp

weather.Year = weather.Year.astype(int) # Making all numeric columns to int
weather.Temp = weather.Temp.astype(float) # Making all columns to float

weather

    gdpNum = float(j[1][:-9])
    gdp.at[i, 'Year'] = year
    gdp.at[i, 'GDP'] = gdpNum

gdp = gdp.drop([92])

gdp.GDP = gdp.GDP.astype(float) # Making all columns to float
